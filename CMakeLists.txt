CMAKE_MINIMUM_REQUIRED(VERSION 2.6)
project(minimal-opencl-gl4)

FIND_PACKAGE (OpenCL)
FIND_PACKAGE(OpenGL)
FIND_PACKAGE(GLUT)
FIND_PACKAGE(GLEW)

INCLUDE_DIRECTORIES( 
    ${GLUT_INCLUDE_DIR}
    ${OPENGL_INCLUDE_DIR}
    ${GLEW_INCLUDE_PATH}
)

ADD_EXECUTABLE(minimal-opencl-gl4 main.cpp cll.cpp part2.cpp util.cpp)

TARGET_LINK_LIBRARIES (minimal-opencl-gl4
   ${GLUT_LIBRARIES}
   ${OPENGL_LIBRARIES}
   ${GLEW_LIBRARY}
   ${OpenCL_LIBRARY}
)
